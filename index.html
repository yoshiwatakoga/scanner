    h2 { color: #1a73e8; margin-top: 0; }
    #canvas { width: 100%; max-width: 400px; border: 3px solid #1a73e8; border-radius: 8px; display: none; margin: 10px auto; }
    .btn { color: white; padding: 20px; font-size: 1.3rem; border: none; border-radius: 10px; width: 100%; margin-top: 15px; cursor: pointer; font-weight: bold; }
    #cameraBtn { background: #28a745; }
    #nextBtn { background: #1a73e8; }

<div class="container">
  <h2>QR在庫管理スキャナー</h2>
  <div id="status">現在の入力先: A列 (1)</div>
  <button id="cameraBtn" onclick="startCamera()" class="btn">📷 スキャン開始</button>
  <canvas id="canvas"></canvas>
  <button id="nextBtn" onclick="moveToNextColumn()" class="btn">➡ 次の品目（列）へ移動</button>
  <div id="log" style="margin-top:15px; font-weight:bold; color:#28a745;"></div>
</div>

<script>
  const GAS_URL = "★ここにGASの新しいURLを貼り付ける★";

  const video = document.createElement("video");
  const canvasElement = document.getElementById("canvas");
  const canvas = canvasElement.getContext("2d");
  const logDiv = document.getElementById("log");
  const statusDiv = document.getElementById("status");
  let currentColumn = 1;
  let lastReadCode = "";

  async function startCamera() {
    logDiv.innerHTML = "カメラを起動しています...";
    try {
      const stream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: "environment" }, audio: false });
      video.srcObject = stream;
      video.setAttribute("playsinline", true);
      await video.play();
      document.getElementById("cameraBtn").style.display = "none";
      canvasElement.style.display = "block";
      requestAnimationFrame(tick);
      logDiv.innerHTML = "✓ バーコード・QRを映してください。";
    } catch (err) {
      alert("ブラウザのカメラ使用を許可してください。");
      logDiv.innerHTML = "❌ 起動失敗: " + err.name;
    }
  }

  function tick() {
    if (video.readyState === video.HAVE_ENOUGH_DATA) {
      canvasElement.height = video.videoHeight;
      canvasElement.width = video.videoWidth;
      canvas.drawImage(video, 0, 0, canvasElement.width, canvasElement.height);
      const imageData = canvas.getImageData(0, 0, canvasElement.width, canvasElement.height);
      const code = jsQR(imageData.data, imageData.width, imageData.height, { inversionAttempts: "dontInvert" });

      if (code && code.data !== lastReadCode) {
        lastReadCode = code.data;
        logDiv.style.color = "#1a73e8";
        logDiv.innerHTML = "シートに書き込み中...";

        // Googleの箱の外から、安全にデータだけをスプレッドシートに叩き込む
        fetch(GAS_URL, {
          method: "POST",
          body: JSON.stringify({ qrValue: code.data, col: currentColumn })
        })
        .then(() => {
          logDiv.style.color = "#28a745";
          logDiv.innerHTML = "✓ 保存完了: " + code.data;
        })
        .catch(() => {
          logDiv.style.color = "#dc3545";
          logDiv.innerHTML = "❌ 送信失敗";
        });

        setTimeout(() => { lastReadCode = ""; }, 3000);
      }
    }
    requestAnimationFrame(tick);
  }

  function moveToNextColumn() {
    currentColumn++;
    statusDiv.innerHTML = "現在の入力先: " + String.fromCharCode(64 + currentColumn) + "列 (" + currentColumn + ")";
  }
</script>
